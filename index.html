# كود هيكلي بسيط لبوت تعدين وهمي (Python) from telegram import Update from telegram.ext import ApplicationBuilder, CommandHandler, ContextTypes import time # قاعدة بيانات وهمية في الذاكرة (استبدلها بـ SQLite لاحقاً) users_db = {} async def start(update: Update, context: ContextTypes.DEFAULT_TYPE): await update.message.reply_text("مرحباً! استخدم /mine لبدء التعدين و /balance لرصيدك.") async def mine(update: Update, context: ContextTypes.DEFAULT_TYPE): user_id = update.effective_user.id current_time = time.time() # التحقق مما إذا كان المستخدم قد عدن سابقاً last_mine = users_db.get(user_id, 0) if current_time - last_mine > 3600: # انتظار ساعة واحدة (3600 ثانية) users_db[user_id] = current_time await update.message.reply_text("تمت إضافة 10 عملات PG إلى رصيدك!") else: await update.message.reply_text("يجب الانتظار ساعة قبل التعدين مرة أخرى.") # إعداد البوت app = ApplicationBuilder().token("YOUR_TOKEN_HERE").build() app.add_handler(CommandHandler("start", start)) app.add_handler(CommandHandler("mine", mine)) app.run_polling()
