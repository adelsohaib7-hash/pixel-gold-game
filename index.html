import time # قاموس لتخزين بيانات المستخدمين (للتجربة فقط، استخدم قاعدة بيانات في الواقع) users_data = {} def start_mining(user_id): # تسجيل وقت بدء التعدين users_data[user_id] = {'start_time': time.time(), 'is_mining': True} return "تم بدء التعدين! عد بعد فترة لاستلام أرباحك." def claim_mining(user_id): if user_id not in users_data or not users_data[user_id]['is_mining']: return "يجب أن تبدأ التعدين أولاً." # حساب الوقت المنقضي elapsed_time = time.time() - users_data[user_id]['start_time'] # معادلة وهمية: (كل ثانية تعطي 0.1 عملة) earned = elapsed_time * 0.1 users_data[user_id]['is_mining'] = False return f"لقد قمت بتعدين {round(earned, 2)} عملة!"
