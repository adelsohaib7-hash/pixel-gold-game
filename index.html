// ملف server.js (يحتاج Node.js) const express = require('express'); const app = express(); const db = {}; // هنا ستضع قاعدة بيانات حقيقية مثل Firebase لاحقاً app.use(express.json()); // حفظ رصيد المستخدم app.post('/update-balance', (req, res) => { const { userId, amount } = req.body; db[userId] = (db[userId] || 0) + amount; res.json({ success: true, balance: db[userId] }); }); app.listen(3000, () => console.log('السيرفر يعمل على بورت 3000'));
